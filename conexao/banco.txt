CREATE DATABASE yftestessistemas;

USE yftestessistemas

// COPIE O CÓDIGO ABAIXO E COLAR NO EDITOR

CREATE TABLE caddesp(

    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    categoria VARCHAR(100) NOT NULL,
    valor DECIMAL(10,2) NOT NULL,
    vencimento DATE NOT NULL,
    formapag VARCHAR(100) NOT NULL,
    imovelassoc VARCHAR(100) NOT NULL,
    parcela INT NOT NULL,
    pago TINYINT NULL DEFAULT(0),
    data_pagamento DATE NOT NULL,
    infocomp TEXT NULL
);

CREATE TABLE cadrec(

    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    tiporec VARCHAR(100) NOT NULL,
    tiporecebe VARCHAR(100) NOT NULL,
    valorrec DECIMAL(10,2) NOT NULL,
    validade DATE NOT NULL,
    recebido TINYINT NULL DEFAULT(0),
    repete INT NOT NULL,
    data_recebimento DATE NOT NULL,
    infocomp TEXT NULL
);

CREATE TABLE cadorc(

    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    titulo VARCHAR(255) NOT NULL,
    validade DATE NOT NULL,
    valorOrc DECIMAL(10,2) NOT NULL,
    valorAtual DECIMAL(10,2) NOT NULL,
    prioridade varchar(20) NULL,
    infocomp TEXT NULL

);
